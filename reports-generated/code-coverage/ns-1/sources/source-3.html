 internal inline fun <T> Iterable<T>.getIndex(index: Int): T {
     return when (this) {
         is List ->
             if (size >= index - 1) this[index - 1]
             else throw NoSuchElementException("Collection size is lower than $index.")
         else -> {
             val iterator = iterator()
             if (!iterator.hasNext())

     }
